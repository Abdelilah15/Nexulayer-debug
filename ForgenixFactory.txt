pragma solidity ^0.8.34;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/token/ERC721/ERC721.sol";

// --- 1. LE MODÈLE DE MESSAGE ---
contract ForgenixMessage {
    string public message;
    address public owner;

    constructor(string memory _message, address _owner) {
        message = _message;
        owner = _owner;
    }
}

// --- 2. LE MODÈLE DE CONTRAT SIMPLE ---
contract ForgenixSimpleContract {
    string public name;
    address public owner;

    constructor(string memory _name, address _owner) {
        name = _name;
        owner = _owner;
    }
}

// --- 3. LES MODÈLES DE TOKENS ERC-20 ---
contract ForgenixToken is ERC20 {
    address public owner;

    constructor(string memory name, string memory symbol, uint256 initialSupply, address _owner) ERC20(name, symbol) {
        owner = _owner;
        _mint(_owner, initialSupply * 10 ** decimals());
    }
}

contract ForgenixAdvancedToken is ERC20 {
    string public contractURI;
    address public owner;

    constructor(
        string memory name,
        string memory symbol,
        uint256 initialSupply,
        string memory _contractURI,
        address _owner
    ) ERC20(name, symbol) {
        contractURI = _contractURI;
        owner = _owner;
        _mint(_owner, initialSupply * 10 ** decimals());
    }
}

// --- 4. LES MODÈLES DE NFT ERC-721 ---
contract ForgenixNFT is ERC721 {
    uint256 public immutable maxSupply;

    constructor(string memory name, string memory symbol, uint256 supply, address _owner) ERC721(name, symbol) {
        maxSupply = supply;
        for (uint256 i = 1; i <= supply; i++) {
            _mint(_owner, i);
        }
    }
}

contract ForgenixAdvancedNFT is ERC721 {
    string private _baseTokenURI;
    uint256 public immutable maxSupply;

    constructor(
        string memory name,
        string memory symbol,
        uint256 supply,
        string memory tokenURI_,
        address _owner
    ) ERC721(name, symbol) {
        _baseTokenURI = tokenURI_;
        maxSupply = supply;
        for (uint256 i = 1; i <= supply; i++) {
            _mint(_owner, i);
        }
    }

    function tokenURI(uint256 /* tokenId */) public view override returns (string memory) {
        return _baseTokenURI;
    }
}

// ==========================================
// --- L'USINE PRINCIPALE (VOTRE PRODUIT) ---
// ==========================================

contract ForgenixFactory {
    address payable public immutable ownerFeeAddress;
    uint256 public constant STANDARD_FEE = 0.00003 ether;
    uint256 public constant ADVANCED_FEE = 0.0002 ether;

    event ContractCreated(address indexed contractAddress, string contractType, address indexed creator);

    constructor() {
        ownerFeeAddress = payable(msg.sender);
    }

    function _takeFee(uint256 requiredFee) internal {
        require(msg.value >= requiredFee, "Frais de service insuffisants");
        (bool success, ) = ownerFeeAddress.call{value: msg.value}("");
        require(success, "Erreur lors du transfert de la commission");
    }

    // SERVICE 1 : Déployer un Message (Standard)
    function createMessage(string memory text) external payable returns (address) {
        _takeFee(STANDARD_FEE);
        ForgenixMessage newContract = new ForgenixMessage(text, msg.sender);
        emit ContractCreated(address(newContract), "Message", msg.sender);
        return address(newContract);
    }

    // SERVICE 2 : Déployer un Contrat Simple (Standard)
    function createSimpleContract(string memory name) external payable returns (address) {
        _takeFee(STANDARD_FEE);
        ForgenixSimpleContract newContract = new ForgenixSimpleContract(name, msg.sender);
        emit ContractCreated(address(newContract), "SimpleContract", msg.sender);
        return address(newContract);
    }

    // SERVICE 3 : Déployer un Token Standard
    function createToken(string memory name, string memory symbol, uint256 supply) external payable returns (address) {
        _takeFee(STANDARD_FEE);
        ForgenixToken newContract = new ForgenixToken(name, symbol, supply, msg.sender);
        emit ContractCreated(address(newContract), "ERC20", msg.sender);
        return address(newContract);
    }

    // SERVICE 4 : Déployer un Token Avancé (Utilise ADVANCED_FEE)
    function createAdvancedToken(
        string memory name,
        string memory symbol,
        uint256 supply,
        string memory _contractURI
    ) external payable returns (address) {
        _takeFee(ADVANCED_FEE); // 0.0002 ETH
        ForgenixAdvancedToken newContract = new ForgenixAdvancedToken(name, symbol, supply, _contractURI, msg.sender);
        emit ContractCreated(address(newContract), "AdvancedERC20", msg.sender);
        return address(newContract);
    }

    // SERVICE 5 : Déployer un NFT Standard
    function createNFT(string memory name, string memory symbol, uint256 supply) external payable returns (address) {
        _takeFee(STANDARD_FEE);
        ForgenixNFT newContract = new ForgenixNFT(name, symbol, supply, msg.sender);
        emit ContractCreated(address(newContract), "ERC721", msg.sender);
        return address(newContract);
    }

    // SERVICE 6 : Déployer un NFT Avancé (Utilise ADVANCED_FEE)
    function createAdvancedNFT(
        string memory name,
        string memory symbol,
        uint256 supply,
        string memory tokenURI
    ) external payable returns (address) {
        _takeFee(ADVANCED_FEE); // 0.0002 ETH
        ForgenixAdvancedNFT newContract = new ForgenixAdvancedNFT(name, symbol, supply, tokenURI, msg.sender);
        emit ContractCreated(address(newContract), "AdvancedERC721", msg.sender);
        return address(newContract);
    }
}
